int MLS1(int seed)
{
    int x0 = BIT_VALUE(seed, 0);
    int x1 = BIT_VALUE(seed, 1);
    int x4 = BIT_VALUE(seed, 4);
    int x5 = BIT_VALUE(seed, 5);
    int w = x0 ^ x1 ^ x4 ^ x5;
    int wynik = (seed >> 1) + (w << 5);
    return wynik;
}

int MLS2(int seed)
{
    int x0 = BIT_VALUE(seed, 0);
    int x3 = BIT_VALUE(seed, 3);
    int x4 = BIT_VALUE(seed, 4);
    int x5 = BIT_VALUE(seed, 5);
    int w = x0 ^ x3 ^ x4 ^ x5;
    int wynik = (seed >> 1) + (w << 5);
    return wynik;
}

int S_Golda(int seed)
{
    return MLS1(seed) ^ MLS2(seed);

}
